function custom_menus() {
 register_nav_menus(
  array(
   'primary' => 'Primary menu',
   'secondary' => 'Secondary menu',
   'tertiary' => 'Tertiary menu'
  )
 );
}

add_action( 'init', 'custom_menus' );
