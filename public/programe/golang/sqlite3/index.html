func dbConn() (db *sql.DB) {
	db, err := sql.Open("sqlite3", dbPath+"?_busy_timeout=5000")
	if err != nil {
		panic(err.Error())
	}
	db.SetConnMaxLifetime(5 * time.Second)
	db.SetMaxIdleConns(0)
	db.SetMaxOpenConns(20)
	return db
}
