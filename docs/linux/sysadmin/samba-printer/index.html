[printers]
   comment = All Printers
   browseable = no
   path = /var/spool/samba
   printable = yes 
   guest ok = no
   read only = yes 
   create mask = 0700
   valid users = dayin

[print$]
   comment = Printer Drivers
   path = /var/lib/samba/printers
   browseable = yes 
   read only = yes 
   guest ok = no
   valid users = dayin
